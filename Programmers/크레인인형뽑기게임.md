## 크레인 인형뽑기 게임 문제
- 문제 url : https://programmers.co.kr/learn/courses/30/lessons/64061

``` cpp
#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, vector<int> moves) {
    int answer = 0;
    vector<int> urn;

    for (int i = 0; i < moves.size(); i++) {
        int line = moves[i] - 1;
        printf("%d\n", line);
        int target;
        for (int i = 0; i < board.size(); i++) {
            if (board[i][line] != 0) {
                    target = board[i][line];
                    board[i][line] = 0;
                    printf("target: %d\n", target);                    
                    if (urn.size() == 0) {
                        urn.push_back(target);
                    }
                    else {
                        if (urn[urn.size()-1] == target) {
                            urn.pop_back();
                            answer+=2;
                        } 
                        else {
                            urn.push_back(target);
                        }
                    }
                    printf("바구니에: %d, answer: %d\n", urn.size(), answer);
                    break;
            }
        }
    }
    return answer;
}
```
